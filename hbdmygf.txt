<script>
// Steps
const steps = [
  {icon:'✨',title:'happy birthday my girlfriend!!',text:"there’s more waiting… tap to find out ✨",cta:'tap'},
  {icon:'✉️',title:'you get more amazing every year. happy birthday, baby.',text:'i wrote you a little something—open it?',cta:'open it'},
  {icon:'💖',title:"you’re already the birthday girl, did you have to be this pretty too? not fair.",text:'okay but seriously… there’s more ↓',cta:'next'},
  {icon:'🎁',title:'okay okay last one—i made something for your bday, read it slowly yaa',text:'ready?',cta:'okaay'}
];
let idx = 0;
const stepIcon = document.getElementById('stepIcon');
const stepCounter = document.getElementById('stepCounter');
const stepTitle = document.getElementById('stepTitle');
const stepText = document.getElementById('stepText');
const stepBtn = document.getElementById('stepBtn');
const resetBtn = document.getElementById('resetBtn');

function renderStep(){
  const s = steps[idx];
  stepIcon.textContent = s.icon;
  stepCounter.textContent = `Step ${idx+1} / ${steps.length}`;
  stepTitle.textContent = s.title;
  stepText.textContent = s.text;
  stepBtn.textContent = s.cta;
  if(idx === steps.length-1){ stepBtn.textContent = 'celebrate 🎉'; }
  resetBtn.hidden = idx===0;
}
stepBtn.addEventListener('click',()=>{
  if(idx < steps.length-1){ idx++; renderStep(); }
  else { burstConfetti(); }
});
resetBtn.addEventListener('click',()=>{ idx=0; renderStep(); });

renderStep();

// Floating hearts
const floating = document.getElementById('floating');
for(let i=0;i<14;i++){
  const span = document.createElement('div');
  span.className='heart';
  span.style.left = (i*7)%100 + '%';
  span.style.bottom = (-10 - i*8) + 'px';
  span.style.fontSize = (12 + (i%6)*4) + 'px';
  span.style.animationDelay = (i*0.6)+'s';
  span.textContent='♡';
  floating.appendChild(span);
}

// Confetti
function burstConfetti(){
  if(window.confetti){ window.confetti({particleCount:140,spread:80,origin:{y:0.6}}); return; }
  const s = document.createElement('script');
  s.src = "https://cdn.jsdelivr.net/npm/canvas-confetti@1.6.0/dist/confetti.browser.min.js";
  s.onload = ()=>{ if(window.confetti) window.confetti({particleCount:140,spread:80,origin:{y:0.6}}); };
  document.head.appendChild(s);
}

// Music control
const audio = document.getElementById('bg-music');
const btn = document.createElement('button');
btn.textContent = '🔊 pause';
btn.style.position='fixed';btn.style.bottom='16px';btn.style.right='16px';btn.style.zIndex=100;btn.style.padding='6px 10px';btn.style.borderRadius='8px';btn.style.border='none';btn.style.background='rgba(225,29,72,.85)';btn.style.color='white';btn.style.fontWeight='700';btn.style.cursor='pointer';
btn.onclick = ()=>{ if(audio.paused){ audio.play(); btn.textContent='🔊 pause'; } else { audio.pause(); btn.textContent='🔈 play'; } };
document.body.appendChild(btn);
</script>
